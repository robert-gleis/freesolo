# Factory Knowledge Base Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Load repo operational knowledge from `.issueflow/knowledge/*.md` and inject it into agent startup prompts at every spawn path (`issueflow start` and workflow engine `spawn`).

**Architecture:** Pure loader module (`src/knowledge/loader.ts`) with filesystem reads at spawn time (no cache). `appendKnowledgeToPrompt` composes the workflow kernel with a formatted knowledge section. Both `createStartPlan` and `createWorkflowEngine` call the loader through injectable deps for testability.

**Tech Stack:** TypeScript (ES2022, NodeNext), Vitest, Node `fs/promises`.

```ts
// src/knowledge/loader.ts
import path from 'node:path';

export interface KnowledgeEntry {
  filename: string;
  title: string;
  content: string;
}

const HEADING_RE = /^#\s+(.+)$/m;

/** Exported for unit tests; not part of the spec Public API surface. */
export function extractTitle(filename: string, content: string): string {
  const match = content.match(HEADING_RE);
  if (match?.[1]) {
    return match[1].trim();
  }

  return path.basename(filename, path.extname(filename));
}

export function formatKnowledgeSection(entries: KnowledgeEntry[]): string {
  if (entries.length === 0) {
    return '';
  }

  const blocks = entries.map((entry) => `### ${entry.title}\n\n${entry.content.trim()}`);

  return [
    '## Factory Knowledge Base',
    '',
    'The following operational knowledge applies to this repository. It is loaded from `.issueflow/knowledge/*.md` at agent spawn time.',
    '',
    blocks.join('\n\n')
  ].join('\n');
}

export function appendKnowledgeToPrompt(basePrompt: string, entries: KnowledgeEntry[]): string {
  const section = formatKnowledgeSection(entries);
  if (!section) {
    return basePrompt;
  }

  return `${basePrompt}\n\n${section}`;
}
```

```ts
export async function loadKnowledgeEntries(repoRoot: string): Promise<KnowledgeEntry[]> {
  const knowledgeDir = path.join(repoRoot, '.issueflow', 'knowledge');

  let names: string[];
  try {
    names = await fs.readdir(knowledgeDir);
  } catch (error) {
    if ((error as NodeJS.ErrnoException).code === 'ENOENT') {
      return [];
    }
    throw error;
  }

  const markdownFiles = names.filter((name) => name.endsWith('.md')).sort();

  const entries: KnowledgeEntry[] = [];

  for (const filename of markdownFiles) {
    const absolutePath = path.join(knowledgeDir, filename);
    const stat = await fs.stat(absolutePath);
    if (!stat.isFile()) {
      continue;
    }

    const content = await fs.readFile(absolutePath, 'utf8');
    entries.push({
      filename,
      title: extractTitle(filename, content),
      content
    });
  }

  return entries;
}
```
